// ── メモ解析 ──

function parseMemo(){

  const raw=document.getElementById('memo-input').value.trim();

  if(!raw)return;

  const lines=raw.split('\n').filter(l=>l.trim());

  let added=0;

  lines.forEach(line=>{const e=parseLine(line.trim());if(e){state.entries.unshift(e);added++;}});

  if(added>0){save();document.getElementById('memo-input').value='';renderEntries();updateTopBalance();}

  else alert('入力を認識できませんでした。\n例: 5/6 コンビニ 980');

}

function parseLine(line){

  const now=new Date();let date=now.toISOString().split('T')[0],text=line;

  const ps=[

    {r:/^(\d{1,2})[\/月](\d{1,2})日?\s+/,fn:m=>`${now.getFullYear()}-${m[1].padStart(2,'0')}-${m[2].padStart(2,'0')}`},

    {r:/^(\d{4})[\/\-](\d{1,2})[\/\-](\d{1,2})\s+/,fn:m=>`${m[1]}-${m[2].padStart(2,'0')}-${m[3].padStart(2,'0')}`},

    {r:/^(\d{1,2})日\s+/,fn:m=>`${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${m[1].padStart(2,'0')}`}

  ];

  for(const p of ps){const m=text.match(p.r);if(m){date=p.fn(m);text=text.slice(m[0].length);break;}}

  const am=text.match(/[,，]?\s*(\d[\d,，]*)\s*円?$/);

  if(!am)return null;

  const amount=parseInt(am[1].replace(/[,，]/g,''));

  const memo=text.slice(0,text.lastIndexOf(am[0])).trim()||text;

  return{id:Date.now()+Math.random(),date,memo,amount,category:guessCategory(memo)};

}

const KW={食費:['スーパー','食料','食品','米','野菜','肉','魚','お菓子','コーヒー'],交通費:['電車','バス','タクシー','交通','suica','pasmo','定期','新幹線'],外食:['ランチ','ディナー','コンビニ','カフェ','レストラン','食事','飲み','ラーメン','寿司','焼肉','居酒屋','マック','スタバ'],日用品:['日用品','ドラッグ','薬局','シャンプー','洗剤','消耗品'],娯楽:['映画','ゲーム','本','漫画','音楽','動画','netflix','spotify'],医療:['病院','クリニック','薬','医療','歯医者'],衣類:['服','靴','洋服','ユニクロ','zara','アパレル']};

function guessCategory(memo){

  const l=memo.toLowerCase();

  for(const[cat,kws]of Object.entries(KW)){if(state.categories.includes(cat)&&kws.some(k=>l.includes(k)))return cat;}

  return state.categories[state.categories.length-1]||'その他';

}
